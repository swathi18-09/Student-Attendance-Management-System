private static void displayAttendance(String askedDate, String[] names, int[] rollNumbers, Map<String, char[][]> attendanceMap, int totalSubjects) {
char[][] attendance = attendanceMap.get(askedDate);
if (attendance == null) {
System.out.println("No attendance data found for the specified date.");
return;
}

System.out.println("\nAttendance Details for " + askedDate + " \n");

// Displaying header
System.out.printf("%-15s %-15s", "Name", "Roll Number");
for (int i = 1; i <= totalSubjects; i++) {
System.out.printf(" %-15s", "Subject " + i);
}
System.out.println(" Attendance %");
System.out.println();

// Displaying student details for the asked date
for (int i = 0; i < names.length; i++) {

System.out.printf("%-15s %-15d", names[i], rollNumbers[i]);

int presentCount = 0;
for (int j = 0; j < totalSubjects; j++) {
System.out.printf(" %-15c", attendance[i][j]);
if (attendance[i][j] == 'P') {
presentCount++;
}
}

// Calculate attendance percentage
double attendancePercentage = (double) presentCount / totalSubjects * 100;
System.out.printf(" %-15.2f", attendancePercentage);

// Display warning message if attendance is less than 65%
if (attendancePercentage < 65) {
System.out.println(" Warning Low Attendance!");
} else {
System.out.println();
}
}
